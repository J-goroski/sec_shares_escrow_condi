# SEC EDGAR pipeline — Python dependencies
# Python 3.10+  (tested on 3.11.9)
#
#     pip install -r requirements.txt
#
# Only three packages are needed for the core methods under methods/.
# Flask (for the optional viewer/) and Jupyter (for the notebook) are separate.

# ── Core (methods/) ───────────────────────────────────────────────────────────
pandas>=2.0          # every module returns/consumes DataFrames
requests>=2.28       # all SEC HTTP access
lxml>=5.0            # XBRL (etree) + HTML parsing; also backs pandas.read_html

# ── Viewer only (viewer/app.py) ───────────────────────────────────────────────
flask>=3.0           # the EDGAR Research Desk app
markupsafe>=2.1      # imported directly by the viewer (also a Flask dependency)

# ── Analysis (analysis/) ──────────────────────────────────────────────────────
# The analysis core needs no extra packages (Ollama over HTTP via `requests`,
# stdlib sqlite3 for the research store).  The local LLM has two
# interchangeable backends (env LLM_BACKEND = auto | ollama | embedded | none;
# without either backend the deterministic regex/rules layers still run):
#
#   A) Ollama server:  install https://ollama.com then `ollama pull gemma2:2b`
#   B) EMBEDDED (serverless — no daemon, nothing to connect to): the two
#      lines below + downloading a weights file once:
#          python analysis/models/download_model.py
#      >>> The weights file lives at:  analysis/models/*.gguf  <<<
#      (currently gemma-2-2b-it-Q4_K_M.gguf, ~1.7 GB, gitignored — safe to
#       delete; re-fetch any time with the command above.)
#
# The extra index below serves prebuilt CPU wheels for llama-cpp-python —
# WITHOUT it, plain PyPI ships source only and pip needs a C++ toolchain
# (fails on this machine).  Comment out both lines if you only use Ollama.
--prefer-binary
--extra-index-url https://abetlen.github.io/llama-cpp-python/whl/cpu
llama-cpp-python>=0.3    # embedded serverless LLM backend (loads the .gguf)
